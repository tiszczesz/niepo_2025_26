Praca z obiektami w JS

Obiekty tworzone literalnie

        const person = {
            name: 'John',
            age: 30,
            greet: function() {
                console.log(`Hello, my name is ${this.name} and I am ${this.age} years old.`);
            }
        };

Obiekty tworzone za pomocą konstruktora

        function Person(name, age) {
            this.name = name;
            this.age = age;
            this.greet = function() {
                console.log(`Hello, my name is ${this.name} and I am ${this.age} years old.`);
            };
        }
        // Tworzenie instancji obiektu
        const person1 = new Person('Alice', 25);
        const person2 = new Person('Bob', 35);

Obiekty tworzone za pomocą klasy

        class Person {
            // Konstruktor klasy wywołany podczas tworzenia instancji czyli obiektu
            // Konstruktor jest specjalną metodą, która jest wywoływana podczas tworzenia instancji klasy.
            // musi nazywać się "constructor" w php jest to "__construct" w c++ jest Person() czyli nazwa klasy
            constructor(name, age) {
                this.name = name;
                this.age = age;
            }
            greet() {
                console.log(`Hello, my name is ${this.name} and I am ${this.age} years old.`);
            }
        }
        // Tworzenie instancji obiektu
        const person3 = new Person('Charlie', 28);
        const person4 = new Person('Diana', 32);
